def initialize(name, extension = false)
	if extension
		if extension == true
			offset = name.rindex('.')
		else
			offset = name.rindex(extension) - 1
		end
		
		@name = name[0...offset]
		@extension = name[offset+1..-1]
	else
		@name = name
		@extension = nil
	end
end
